
#include<avr/interrupt.h>
#include<avr/io.h>
char x[5]={"VNSGU"};
char y[5]={"  INDIA"};
char a,b;
int i=0,j;
void setup() {
  // put your setup code here, to run once:
  DDRB=0x20;
  UCSR0A=0x00;
  UCSR0B=0xD8;
  UBRR0=0x67;
  //UDR0=x;
  i=0;
  j=0;
}

void loop() {
  // put your main code here, to run repeatedly:
   //UDR0=a;
   
  if(a=='f')
  {
    for(i;i<5;i++)
    {
      UDR0=x[i];
      delay(1000);
    }
  }
  if(a=='r')
  {
    for(j;j<5;j++)
    {
      UDR0=y[j];
      delay(1000);
    }
  }
  
}
ISR(USART_TX_vect)
{
 
}
ISR(USART_RX_vect)
{
   
 a=UDR0;
 i=0;
 j=0;
}
